program MenuOpciones;
var
  opcion: integer;
begin
  repeat
    writeln('=== MENÚ DE OPCIONES ===');
    writeln('1. Saludar');
    writeln('2. Sumar dos números');
    writeln('3. Salir');
    writeln('Seleccione una opción:');
    readln(opcion);

    case opcion of
      1: writeln('¡Hola, usuario!');
      2: begin
        var a, b: integer;
        writeln('Ingrese dos números:');
        readln(a, b);
        writeln('La suma es: ', a + b);
      end;
      3: writeln('Saliendo...');
    else
      writeln('Opción inválida. Intente de nuevo.');
    end;
  until opcion = 3;
end.
