program TorresDeHanoi;

procedure MoverDiscos(n: integer; desde, hacia, auxiliar: char);
begin
  if n = 1 then
    writeln('Mover disco de ', desde, ' a ', hacia)
  else
  begin
    MoverDiscos(n - 1, desde, auxiliar, hacia);
    writeln('Mover disco de ', desde, ' a ', hacia);
    MoverDiscos(n - 1, auxiliar, hacia, desde);
  end;
end;

var
  n: integer;
begin
  writeln('Ingrese el n√∫mero de discos:');
  readln(n);
  MoverDiscos(n, 'A', 'C', 'B');
end.
